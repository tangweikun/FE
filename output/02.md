```js
function mo() {
  var x = 0;
  return function() {
    console.log(++x);
  };
}
var a = mo();
var b = mo();
a();
a();
b();

// 1 2 1
// a 和 b是不同的两个函数，分别保持着各自对不同的x的引用
```

```js
var p = [];
var A = new Function();
A.prototype = p;
var a = new A();
a.push(1);
console.log(a.length);
console.log(p.length);

// 1 0
// new 如果在继承对象是没有参数的情况下，是可以不加后面的括号的，编译器会自动替你加上的。但是要记住new的没有括号优先级是低于`.`的，有括号是高于`()`;
```
